wire [{word_size}-1:0] acc_output;
/* n_parity_sym_nr can be 1 bit */
/* verilator lint_off ASCRANGE */
reg [$clog2({n_parity_max})-1:0] n_parity_sym_nr = 0;
/* verilator lint_on ASCRANGE */
reg feedback_control_reg = 0;

always @(posedge aclk) begin : feedback_control
    casez({{feedback_control_reg, axis_s_tlast, areset_n}})
        3'b?11: begin
            feedback_control_reg <= 1'b1;
            n_parity_sym_nr <= {n_parity_max};
        end
        3'b101: begin
            if (n_parity_sym_nr == 0) begin
                feedback_control_reg <= 1'b0;
            end else begin
                n_parity_sym_nr <= n_parity_sym_nr - 1;
            end
        end
        default: begin
            feedback_control_reg <= 0;
            n_parity_sym_nr <= 0;
        end
    endcase
end // always

assign axis_s_tready = !feedback_control_reg;
assign axis_m_tvalid = |n_parity_sym_nr ? 1'b1 : (axis_s_tvalid && ~feedback_control_reg);
assign axis_m_tdata = feedback_control_reg ? acc_output : axis_s_tdata;
assign axis_m_tlast = n_parity_sym_nr==1; //last parity word
